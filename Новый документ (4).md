import sys

from core.acp\_core\_v1\_0 import ACPCore  
from epms\_templates.bearing\_epms import map\_bearing

def run\_demo():  
    acp \= ACPCore()  
      
    \# Имитация потока данных от сенсоров  
    telemetry \= \[  
        {"force\_n": 2000, "rpm": 1500, "fatigue\_hours": 1000, "clearance\_mm": 0.01},  
        {"force\_n": 8000, "rpm": 2800, "fatigue\_hours": 1050, "clearance\_mm": 0.04},  
        {"force\_n": 9500, "rpm": 2900, "fatigue\_hours": 1100, "clearance\_mm": 0.05}  
    \]

    print("--- BEARING DEMO \---")  
    for data in telemetry:  
        epms\_vector \= map\_bearing(\*\*data)  
        report \= acp.evaluate(\*\*epms\_vector)  
        print(f"Data: {data} \-\> Report: {report\['state\_name'\]} (Risk: {report\['risk'\]})")

if \_\_name\_\_ \== "\_\_main\_\_":  
    run\_demo()
