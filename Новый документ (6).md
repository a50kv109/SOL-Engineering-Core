import sys  
import os  
sys.path.append(os.path.dirname(os.path.dirname(os.path.abspath(\_\_file\_\_))))

from core.acp\_core\_v1\_0 import ACPCore

def map\_server(cpu\_load, memory\_swap\_rate, uptime\_days, thermal\_margin):  
    return {  
        "effort": min(cpu\_load / 100.0, 1.0),  
        "velocity": min(memory\_swap\_rate / 1000.0, 1.0),  
        "wear": min(uptime\_days / 365.0, 1.0),  
        "stability": min(thermal\_margin / 40.0, 1.0)  
    }

def run\_demo():  
    acp \= ACPCore()  
    print("--- SERVER DEMO \---")  
    epms \= map\_server(cpu\_load=95, memory\_swap\_rate=800, uptime\_days=300, thermal\_margin=5)  
    report \= acp.evaluate(\*\*epms)  
    print(f"Server Report: {report\['state\_name'\]} (Risk: {report\['risk'\]})")

if \_\_name\_\_ \== "\_\_main\_\_":  
    run\_demo()
