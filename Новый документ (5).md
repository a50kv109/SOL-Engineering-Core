import sys  
import os  
sys.path.append(os.path.dirname(os.path.dirname(os.path.abspath(\_\_file\_\_))))

from core.acp\_core\_v1\_0 import ACPCore

def map\_cad(intersection\_depth, change\_rate, edit\_count, constraint\_coverage):  
    return {  
        "effort": min(intersection\_depth / 10.0, 1.0),  
        "velocity": min(change\_rate / 5.0, 1.0),  
        "wear": min(edit\_count / 100.0, 1.0),  
        "stability": constraint\_coverage  
    }

def run\_demo():  
    acp \= ACPCore()  
    print("--- CAD DEMO \---")  
    epms \= map\_cad(intersection\_depth=9.0, change\_rate=0.0, edit\_count=0, constraint\_coverage=0.1)  
    report \= acp.evaluate(\*\*epms)  
    print(f"CAD Conflict Report: {report\['state\_name'\]} (Risk: {report\['risk'\]})")

if \_\_name\_\_ \== "\_\_main\_\_":  
    run\_demo()
